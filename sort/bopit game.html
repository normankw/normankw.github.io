    function start() {
        if (!isplaying) {
            isplaying = true;
            timer = setInterval(function() {
                timeleft--;
                timerleft.innerHTML = timeleft;
                if (timeleft <= 0) {
                    gameover();
                }
            }, 1000);
		pickchoose();
        }
    }
    function pickchoose() {
        var dice = Math.floor(Math.random() * 3) + 1;
        if (1 == dice) {
            choose = "twist";
            chooser.innerHTML = choose;
        } else if (2 == dice) {
            choose = 'bopit';
            chooser.innerHTML = choose;
        } else if (dice == 3) {
            choose = 'pull';
            chooser.innerHTML = choose;
        }
    }

    function resetimer() {
        timeleft = 10;
        clearInterval(timer);
        timer = setInterval(function() {
            timeleft--;
            timerleft.innerHTML = timeleft;
            if (timeleft <= 0) {
                gameover();
            }
        }, 1000);

    }

    function twist() {
        if (isplaying) {
            if (choose == 'twist') {
                score += 5;
                resetimer();
                pickchoose();
                scorer.innerHTML = score;
            } else {
                gameover()
            }
        }
    }

    function bopit() {
        if (isplaying) {

                score += 5;

    function pull() {
