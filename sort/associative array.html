Associative array

This project uses an associative array to find duplicate elements in an array.

back to index


function countDuplicates(list) {
  // using an associative array
  var keyValues = {};
  list.forEach(function(n) {
    if(keyValues[n]) {
         keyValues[n]++;
    } else {
         keyValues[n] = 1;
    }
  }
)  
  return keyValues;
  
}
